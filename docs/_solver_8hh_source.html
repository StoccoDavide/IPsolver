399 bool solve(const Vector& x_guess, Vector& x_sol)
400 {
401 #define CMD "IPsolver::Solver::solve(...): "
402
403 using Array = Eigen::ArrayXd;
404 using Mask = Eigen::Array<bool, Eigen::Dynamic, 1>;
405
406 // INITIALIZATION
407 // Get the number of primal variables (n), the number of constraints (m), the total number of
408 // primal-dual optimization variables (nv), and initialize the Lagrange multipliers and the
409 // second-order information
410 Vector x(x_guess);
411 Vector c(this->m_problem->constraints(x));
412 Integer n{static_cast<Integer>(x.size())};
413 Integer m{static_cast<Integer>(c.size())};
414 Integer nv{n + m};
415 Vector z(Vector::Ones(m));
416 Matrix B(Matrix::Identity(n, n));
417
418 // Repeat while the convergence criterion has not been satisfied, and we haven't reached the
419 // maximum number of iterations
420 Real alpha{0.0};
421 Vector g_old, p_x, p_z;
422 bool converged{false};
423 for (Integer iter{0}; iter < this->m_max_iterations; ++iter)
424 {
425
426 // COMPUTE OBJECTIVE, GRADIENT, CONSTRAINTS, ETC
427 // Compute the response of the objective function, the gradient of the objective, the
428 // response of the inequality constraints, the Jacobian of the inequality constraints, the
429 // Hessian of the Lagrangian (minus the Hessian of the objective) and, optionally, the
430 // Hessian of the objective.
431 Real f{this->m_problem->objective(x)};
432 c = this->m_problem->constraints(x);
433 Vector g(this->m_problem->objective_gradient(x));
434 Matrix J(this->m_problem->constraints_jacobian(x, z));
435 Matrix W(this->m_problem->lagrangian_hessian(x, z));
436 if (this->m_descent == Descent::NEWTON) {
437 B = this->m_problem->objective_hessian(x);
438 }
439
440 // Compute the responses of the unperturbed Karush-Kuhn-Tucker optimality conditions.
441 Vector r_x{g + J.transpose() * z}; // Dual residual
442 Vector r_c{c.array() * z.array()}; // Complementarity
443 Vector r_0(nv);
444 r_0 << r_x, r_c;
445
446 // Set some parameters that affect convergence of the primal-dual interior-point method
447 Real norm_r0{static_cast<Real>(r_0.norm())};
448 Real eta{std::min<Real>(this->m_eta_max, norm_r0 / nv)};
449 Real sigma{std::min<Real>(this->m_sigma_max, std::sqrt(norm_r0 / nv))};
450 Real duality_gap{static_cast<Real>(-c.dot(z))};
451 Real mu{std::max<Real>(this->m_mu_min, sigma * duality_gap / m)};
452
453 // CONVERGENCE CHECK
454 // If the norm of the responses is less than the specified tolerance, we are done
455 if (norm_r0 / nv < this->m_tolerance) {
456 converged = true;
457 break;
458 }
459
460 // Update the BFGS approximation to the Hessian of the objective
461 if (this->m_descent == Descent::BFGS && iter > 0) {
462 B = this->bfgs_update(B, alpha * p_x, g - g_old);
463 }
464
465 // SOLUTION TO PERTURBED KKT SYSTEM
466 // Compute the search direction of x and z
467 Vector c_epsilon(c.array() - this->m_epsilon);
468 Matrix S((z.array() / c_epsilon.array()).matrix().asDiagonal());
469 Vector g_b(g - mu * J.transpose() * (1.0 / c_epsilon.array()).matrix());
470 Matrix H(B + W - J.transpose() * S * J);
471 p_x = H.ldlt().solve(-g_b);
472 p_z = -(z + mu * (1.0 / c_epsilon.array()).matrix() + S * J * p_x);
473
474 // BACKTRACKING LINE SEARCH
475 // To ensure global convergence, execute backtracking line search to determine the step length
476 // First, we have to find the largest step size which ensures that z remains feasible
477 // Next, we perform backtracking line search
478 alpha = this->m_alpha_max;
479 Mask mask((z + p_z).array() < 0.0);
480 if (mask.any()) {
481 Array ratio(z.array() / (-p_z.array()));
482 alpha = this->m_alpha_max * std::min<Real>(1.0, mask.select(ratio, 1.0).minCoeff());
483 }
484
485 // Compute the response of the merit function and the directional gradient at the current
486 // point and search direction
487 Real psi{this->merit(x, z, f, c, mu)};
488 Real dpsi{this->grad_merit(x, z, p_x, p_z, g, c, J, mu)};
489 while (true) {
490
491 // Compute the candidate point, the constraints, and the response of the objective function
492 // and merit function at the candidate point
493 Vector x_new(x + alpha * p_x);
494 Vector z_new(z + alpha * p_z);
495 f = this->m_problem->objective(x_new);
496 c = this->m_problem->constraints(x_new);
497 Real psi_new{this->merit(x_new, z_new, f, c, mu)};
498
499 // Stop backtracking search if we've found a candidate point that sufficiently decreases
500 // the merit function and satisfies all the constraints
501 if ((c.array() <= 0.0).all() && psi_new < psi + this->m_tau * eta * alpha * dpsi) {
502 x = x_new;
503 z = z_new;
504 g_old = g;
505 break;
506 }
507
508 // The candidate point does not meet our criteria, so decrease the step size for 0 < β < 1.
509 alpha *= this->m_beta;
510 if (alpha < this->m_alpha_min) {
511 IPSOLVER_ERROR(CMD "line search step size too small");
512 return false;
513 }
514 }
515 }
516 x_sol = x;
517 return converged;
518
519 #undef CMD
520 }

522private:
539 Real merit([[maybe_unused]] const Vector& x, const Vector& z, Real f, const Vector& c, Real mu)
540 {
541 return f - c.dot(z) - mu * ((c.array().square() * z.array() + this->m_epsilon).log().sum());
542 }
543
559 Real grad_merit([[maybe_unused]] const Vector& x, const Vector& z, const Vector& p_x,
560 const Vector& p_z, const Vector& g, const Vector& c, const Matrix& J, Real mu)
561 {
562 return p_x.dot(
563 g - J.transpose() * z - 2.0 * mu * J.transpose() * (1.0 / (c.array() - this->m_epsilon)).matrix()
564 ) - p_z.dot(
565 c + mu * (1.0 / (z.array() + this->m_epsilon)).matrix()
566 );
567 }

587 Matrix bfgs_update(const Matrix& B, const Vector& s, const Vector& y)
588 {
589 #define CMD "IPsolver::Solver::bfgs_update(...): "
590 IPSOLVER_ASSERT(y.dot(s) > 0.0, CMD "update condition yᵀs > 0 not satisfied");
591 Vector x(B * s);
592 return B - (x * x.transpose()) / (s.dot(x)) + (y * y.transpose()) / (y.dot(s));
593 #undef CMD
594 }
