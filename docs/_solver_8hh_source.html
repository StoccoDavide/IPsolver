157 Vector solve(const Vector& x_guess)
158 {
159 Vector x(x_guess);
160 Vector c(this->m_constraints(x));
161 Integer n{static_cast<Integer>(x.size())};
162 Integer m{static_cast<Integer>(c.size())};
163 Integer nv{n + m};
164
165 Vector z(Vector::Ones(m));
166 Matrix B(Matrix::Identity(n, n));
167
168 Vector g_old;
169 Vector p_x;
170
171 if (this->m_verbose) {
172 std::cout << "i, f(x), lg(mu), sigma, ||r_x||, ||r_c||, alpha, #ls" << std::endl;
173 }
174
175 Real alpha{0.0};
176 Integer ls{0};
177
178 for (Integer iter{1}; iter <= m_max_iterations; ++iter) {
179 Real f{this->m_objective(x)};
180 c = this->m_constraints(x);
181 auto [J, W] = this->m_jacobian(x, z);
182
183 Vector g;
184 if (this->m_descent == Descent::NEWTON) {
185 g = this->m_gradient(x);
186 // B is updated below if BFGS is used
187 } else {
188 g = this->m_gradient(x);
189 }
190
191 Vector r_x{g + J.transpose() * z};
192 Vector r_c{c.array() * z.array()};
193 Vector r_0(nv);
194 r_0 << r_x, r_c;
195
196 Real norm_r0{static_cast<Real>(r_0.norm())};
197
198 Real eta{std::min(this->m_eta_max, norm_r0 / nv)};
199 Real sigma{std::min(this->m_sigma_max, std::sqrt(norm_r0 / nv))};
200 Real duality_gap{static_cast<Real>(-c.dot(z))};
201 Real mu{std::max(this->m_mu_min, sigma * duality_gap / m)};
202
203 if (this->m_verbose) {
204 std::cout << iter << ", " << f << ", " << std::log10(mu) << ", " << sigma << ", "
205 << r_x.norm() << ", " << r_c.norm() << ", " << alpha << ", " << ls << std::endl;
206 }
207
208 if (norm_r0 / nv < this->m_tolerance) {break;}
209
210 if (this->m_descent == Descent::BFGS && iter > 1) {
211 Vector y(g - g_old);
212 B = this->bfgs_update(B, alpha * p_x, y);
213 }
214
215 Vector c_epsilon(c.array() - this->m_epsilon);
216 Matrix S(z.array() / c_epsilon.array());
217 Matrix S_diag(S.asDiagonal());
218
219 Vector g_b(g - mu * J.transpose() * (1.0 / c_epsilon.array()).matrix());
220
221 // Solve (B + W - J' * S * J) * px = -gb
222 Matrix H = B + W - J.transpose() * S_diag * J;
223 p_x = H.ldlt().solve(-g_b);
224
225 Vector p_z(-(z + mu * (1.0 / c_epsilon.array()).matrix() + S_diag * J * p_x));
226
227 alpha = this->m_alpha_max;
228 for (Integer i{0}; i < m; ++i) {
229 if (p_z[i] < 0) {alpha = std::min<Real>(alpha, this->m_alpha_max * (z[i] / (-p_z[i])));}
230 }
231
232 Real psi{this->merit(z, f, c, mu)};
233 Real dpsi{this->grad_merit(z, p_x, p_z, g, c, J, mu)};
234 ls = 0;
235
236 while (true) {
237 ls++;
238 Vector x_new(x + alpha * p_x);
239 Vector z_new(z + alpha * p_z);
240 Real f_new{this->m_objective(x_new)};
241 Vector c_new(this->m_constraints(x_new));
242 Real psi_new{this->merit(z_new, f_new, c_new, mu)};
243
244 bool allFeasible{(c_new.array() <= 0.0).all()};
245 if (allFeasible && psi_new < psi + this->m_tau * eta * alpha * dpsi) {
246 x = x_new;
247 z = z_new;
248 g_old = g;
249 break;
250 }
251 alpha *= this->m_beta;
252 if (alpha < this->m_alpha_min) {
253 IPSOLVER_ERROR("IPsolver::Solver::solve(...): line search step size too small");
254 }
255 }
256 }
257 return x;
258 }

276 Real merit(const Vector& z, Real f, const Vector& c, Real mu)
277 {
278 return f - c.dot(z) - mu * ((c.array().square() * z.array() + this->m_epsilon).log().sum());
279 }
280
295 Real grad_merit(const Vector& z, const Vector& p_x, const Vector& p_z, const Vector& g, const Vector& c,
296 const Matrix& J, Real mu)
297 {
298 Vector term1(g - J.transpose() * z - 2.0 * mu * J.transpose() * (1.0 / (c.array() - this->m_epsilon)).matrix());
299 Vector term2(c + mu * (1.0 / (z.array() + this->m_epsilon)).matrix());
300 return p_x.dot(term1) - p_z.dot(term2);
301 }

321 Matrix bfgs_update(const Matrix& B, const Vector& s, const Vector& y)
322 {
323 if (y.dot(s) <= 0) {IPSOLVER_ERROR("BFGS update condition y^T s > 0 not satisfied");}
324 Vector Bs(B * s);
325 return B - (Bs * Bs.transpose()) / (s.dot(Bs)) + (y * y.transpose()) / (y.dot(s));
326 }
