440 Vector solve(const Vector& x_guess)
441 {
442 #define CMD "IPsolver::Solver::solve(...): "
443
444 IPSOLVER_ASSERT(this->m_objective,
445 CMD "objective function must not be null");
446 IPSOLVER_ASSERT(this->m_objective_gradient,
447 CMD "gradient of the objective function must not be null");
448 IPSOLVER_ASSERT(this->m_objective_hessian || this->m_descent != Descent::NEWTON,
449 CMD "hessian of the objective function must not be null");
450 IPSOLVER_ASSERT(this->m_constraints,
451 CMD "constraints function must not be null");
452 IPSOLVER_ASSERT(this->m_constraints_jacobian,
453 CMD "jacobian of the constraints function must not be null");
454 IPSOLVER_ASSERT(this->m_lagrangian_hessian,
455 CMD "lagrangian hessian function must not be null");
456
457 // INITIALIZATION
458 // Get the number of primal variables (n), the number of constraints (m), the total number of
459 // primal-dual optimization variables (nv), and initialize the Lagrange multipliers and the
460 // second-order information
461 Vector x(x_guess);
462 Vector c(this->m_constraints(x));
463 Integer n{static_cast<Integer>(x.size())};
464 Integer m{static_cast<Integer>(c.size())};
465 Integer nv{n + m};
466 Vector z(Vector::Ones(m));
467 Matrix B(Matrix::Identity(n, n));
468
469 Vector g_old, p_x, p_z;
470 if (this->m_verbose) {
471 std::cout << "i, f(x), lg(mu), sigma, ||r_x||, ||r_c||, alpha, #ls" << std::endl;
472 }
473
474 // Repeat while the convergence criterion has not been satisfied, and we haven't reached the
475 // maximum number of iterations
476 Real alpha{0.0};
477 Integer ls{0};
478 for (Integer iter{0}; iter < m_max_iterations; ++iter) {
479
480 // COMPUTE OBJECTIVE, GRADIENT, CONSTRAINTS, ETC
481 // Compute the response of the objective function, the gradient of the objective, the
482 // response of the inequality constraints, the Jacobian of the inequality constraints, the
483 // Hessian of the Lagrangian (minus the Hessian of the objective) and, optionally, the
484 // Hessian of the objective.
485 Real f{this->m_objective(x)};
486 c = this->m_constraints(x);
487 Vector g(this->m_objective_gradient(x));
488 Matrix J(this->m_constraints_jacobian(x, z));
489 Matrix W(this->m_lagrangian_hessian(x, z));
490 if (this->m_descent == Descent::NEWTON) {
491 B = this->m_objective_hessian(x);
492 }
493
494 // Compute the responses of the unperturbed Karush-Kuhn-Tucker optimality conditions.
495 Vector r_x{g + J.transpose() * z}; // Dual residual
496 Vector r_c{c.array() * z.array()}; // Complementarity
497 Vector r_0(nv);
498 r_0 << r_x, r_c;
499
500 // Set some parameters that affect convergence of the primal-dual interior-point method
501 Real norm_r0{static_cast<Real>(r_0.norm())};
502 Real eta{std::min<Real>(this->m_eta_max, norm_r0 / nv)};
503 Real sigma{std::min<Real>(this->m_sigma_max, std::sqrt(norm_r0 / nv))};
504 Real duality_gap{static_cast<Real>(-c.dot(z))};
505 Real mu{std::max<Real>(this->m_mu_min, sigma * duality_gap / m)};
506
507 // Print the status of the algorithm
508 if (this->m_verbose) {
509 std::cout << iter+1 << ", " << f << ", " << std::log10(mu) << ", " << sigma << ", "
510 << r_x.norm() << ", " << r_c.norm() << ", " << alpha << ", " << ls << std::endl;
511 }
512
513 // CONVERGENCE CHECK
514 // If the norm of the responses is less than the specified tolerance, we are done
515 if (norm_r0 / nv < this->m_tolerance) {break;}
516
517 // Update the BFGS approximation to the Hessian of the objective
518 if (this->m_descent == Descent::BFGS && iter > 0) {
519 B = this->bfgs_update(B, alpha * p_x, g - g_old);
520 }
521
522 // SOLUTION TO PERTURBED KKT SYSTEM
523 // Compute the search direction of x and z
524 Vector c_epsilon(c.array() - this->m_epsilon);
525 Matrix S((z.array() / c_epsilon.array()).matrix().asDiagonal());
526 Vector g_b(g - mu * J.transpose() * (1.0 / c_epsilon.array()).matrix());
527 Matrix H(B + W - J.transpose() * S * J);
528 p_x = H.ldlt().solve(-g_b);
529 p_z = -(z + mu * (1.0 / c_epsilon.array()).matrix() + S * J * p_x);
530
531 // BACKTRACKING LINE SEARCH
532 // To ensure global convergence, execute backtracking line search to determine the step length
533 // First, we have to find the largest step size which ensures that z remains feasible
534 // Next, we perform backtracking line search
535 alpha = this->m_alpha_max;
536 Mask mask((z + p_z).array() < 0.0);
537 if (mask.any()) {
538 Array ratio(z.array() / (-p_z.array()));
539 alpha = this->m_alpha_max * std::min<Real>(1.0, mask.select(ratio, 1.0).minCoeff());
540 }
541
542 // Compute the response of the merit function and the directional gradient at the current
543 // point and search direction
544 Real psi{this->merit(x, z, f, c, mu)};
545 Real dpsi{this->grad_merit(x, z, p_x, p_z, g, c, J, mu)};
546 ls = 0;
547 while (true) {
548
549 // Compute the candidate point, the constraints, and the response of the objective function
550 // and merit function at the candidate point
551 ++ls;
552 Vector x_new(x + alpha * p_x);
553 Vector z_new(z + alpha * p_z);
554 f = this->m_objective(x_new);
555 c = this->m_constraints(x_new);
556 Real psi_new{this->merit(x_new, z_new, f, c, mu)};
557
558 // Stop backtracking search if we've found a candidate point that sufficiently decreases
559 // the merit function and satisfies all the constraints
560 if ((c.array() <= 0.0).all() && psi_new < psi + this->m_tau * eta * alpha * dpsi) {
561 x = x_new;
562 z = z_new;
563 g_old = g;
564 break;
565 }
566
567 // The candidate point does not meet our criteria, so decrease the step size for 0 < β < 1.
568 alpha *= this->m_beta;
569 IPSOLVER_ASSERT(alpha > this->m_alpha_min, CMD "line search step size too small");
570 }
571 }
572 return x;
573
574 #undef CMD
575 }
576
577private:
594 Real merit([[maybe_unused]] const Vector& x, const Vector& z, Real f, const Vector& c, Real mu)
595 {
596 return f - c.dot(z) - mu * ((c.array().square() * z.array() + this->m_epsilon).log().sum());
597 }
598
614 Real grad_merit([[maybe_unused]] const Vector& x, const Vector& z, const Vector& p_x,
615 const Vector& p_z, const Vector& g, const Vector& c, const Matrix& J, Real mu)
616 {
617 return p_x.dot(
618 g - J.transpose() * z - 2.0 * mu * J.transpose() * (1.0 / (c.array() - this->m_epsilon)).matrix()
619 ) - p_z.dot(
620 c + mu * (1.0 / (z.array() + this->m_epsilon)).matrix()
621 );
622 }

642 Matrix bfgs_update(const Matrix& B, const Vector& s, const Vector& y)
643 {
644 #define CMD "IPsolver::Solver::bfgs_update(...): "
645 IPSOLVER_ASSERT(y.dot(s) > 0.0, CMD "update condition yᵀs > 0 not satisfied");
646 Vector x(B * s);
647 return B - (x * x.transpose()) / (s.dot(x)) + (y * y.transpose()) / (y.dot(s));
648 #undef CMD
649 }
