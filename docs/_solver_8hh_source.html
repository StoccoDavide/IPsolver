423 bool solve(const VectorN & x_guess, VectorN & x_sol)
424 {
425 #define CMD "IPsolver::Solver::solve(...): "
426
427 using MatrixM = Eigen::Matrix<Real, M, M>;
428 using ArrayM = Eigen::Array<Real, M, 1>;
429 using MaskM = Eigen::Array<bool, M, 1>;
430
431 // INITIALIZATION
432 // Get the number of primal variables (n), the number of constraints (m), the total number of
433 // primal-dual optimization variables (nv), and initialize the Lagrange multipliers and the
434 // second-order information
435 VectorN x(x_guess);
436 VectorM c;
437 bool success{this->m_problem->constraints(x, c)};
438 IPSOLVER_ASSERT(success, CMD "failed to evaluate constraints");
439 Integer n{static_cast<Integer>(x.size())};
440 Integer m{static_cast<Integer>(c.size())};
441 Integer nv{n + m};
442 VectorM z(VectorM::Ones(m));
443 MatrixH B(MatrixH::Identity(n, n));
444
445 // Repeat while the convergence criterion has not been satisfied, and we haven't reached the
446 // maximum number of iterations
447 Real alpha{0.0}, f, norm_r0, eta, sigma, duality_gap, mu, psi, dpsi, psi_new;
448 VectorN r_x, g_b, g_old, p_x, g, x_new;
449 VectorM r_c, c_epsilon, p_z, z_new;
450 MatrixJ J;
451 MatrixH H, W;
452 MatrixM S;
453 MaskM mask;
454 ArrayM ratio;
455 bool converged{false};
456 for (Integer iter{0}; iter < this->m_max_iterations; ++iter)
457 {
458
459 // COMPUTE OBJECTIVE, GRADIENT, CONSTRAINTS, ETC
460 // Compute the response of the objective function, the gradient of the objective, the
461 // response of the inequality constraints, the Jacobian of the inequality constraints, the
462 // Hessian of the Lagrangian (minus the Hessian of the objective) and, optionally, the
463 // Hessian of the objective.
464 success = this->m_problem->objective(x, f);
465 IPSOLVER_ASSERT(success, CMD "failed to evaluate objective function");
466 success = this->m_problem->constraints(x, c);
467 IPSOLVER_ASSERT(success, CMD "failed to evaluate constraints");
468 success = this->m_problem->objective_gradient(x, g);
469 IPSOLVER_ASSERT(success, CMD "failed to evaluate objective gradient");
470 success = this->m_problem->constraints_jacobian(x, J);
471 IPSOLVER_ASSERT(success, CMD "failed to evaluate constraints jacobian");
472 success = this->m_problem->lagrangian_hessian(x, z, W);
473 IPSOLVER_ASSERT(success, CMD "failed to evaluate lagrangian hessian");
474 if (this->m_descent == Descent::NEWTON) {
475 success = this->m_problem->objective_hessian(x, B);
476 IPSOLVER_ASSERT(success, CMD "failed to evaluate objective hessian");
477 }
478
479 // Compute the responses of the unperturbed Karush-Kuhn-Tucker optimality conditions.
480 r_x = g + J.transpose() * z; // Dual residual
481 r_c = c.array() * z.array(); // Complementarity
482
483 // Set some parameters that affect convergence of the primal-dual interior-point method
484 norm_r0 = std::sqrt<Real>(r_x.squaredNorm() + r_c.squaredNorm());
485 eta = std::min<Real>(this->m_eta_max, norm_r0 / nv);
486 sigma = std::min<Real>(this->m_sigma_max, std::sqrt(norm_r0 / nv));
487 duality_gap = static_cast<Real>(-c.dot(z));
488 mu = std::max<Real>(this->m_mu_min, sigma * duality_gap / m);
489
490 if (this->m_verbose) {
491 std::cout << "Iteration " << iter << std::setprecision(6) << std::scientific
492 << ": f = " << f << ", |r| = " << norm_r0 / nv << std::endl;
493 }
494
495 // CONVERGENCE CHECK
496 // If the norm of the responses is less than the specified tolerance, we are done
497 if (norm_r0 / nv < this->m_tolerance) {
498 converged = true;
499 break;
500 }
501
502 // Update the BFGS approximation to the Hessian of the objective
503 if (this->m_descent == Descent::BFGS && iter > 0) {
504 this->bfgs_update(B, alpha * p_x, g - g_old);
505 }
506
507 // SOLUTION TO PERTURBED KKT SYSTEM
508 // Compute the search direction of x and z
509 c_epsilon = c.array() - this->m_epsilon;
510 S = (z.array() / c_epsilon.array()).matrix().asDiagonal();
511 g_b = g - mu * J.transpose() * (1.0 / c_epsilon.array()).matrix();
512 H = B + W - J.transpose() * S * J;
513 p_x = H.ldlt().solve(-g_b);
514 p_z = -(z + mu * (1.0 / c_epsilon.array()).matrix() + S * J * p_x);
515
516 // BACKTRACKING LINE SEARCH
517 // To ensure global convergence, execute backtracking line search to determine the step length
518 // First, we have to find the largest step size which ensures that z remains feasible
519 // Next, we perform backtracking line search
520 alpha = this->m_alpha_max;
521 mask = (z + p_z).array() < 0.0;
522 if (mask.any()) {
523 ratio = z.array() / (-p_z.array());
524 alpha = this->m_alpha_max * std::min<Real>(1.0, mask.select(ratio, 1.0).minCoeff());
525 }
526
527 // Compute the response of the merit function and the directional gradient at the current
528 // point and search direction
529 psi = this->merit(x, z, f, c, mu);
530 dpsi = this->grad_merit(x, z, p_x, p_z, g, c, J, mu);
531 while (true) {
532
533 // Compute the candidate point, the constraints, and the response of the objective function
534 // and merit function at the candidate point
535 x_new = x + alpha * p_x;
536 z_new = z + alpha * p_z;
537 success = this->m_problem->objective(x_new, f);
538 IPSOLVER_ASSERT(success, CMD "failed to evaluate objective function at candidate point");
539 success = this->m_problem->constraints(x_new, c);
540 IPSOLVER_ASSERT(success, CMD "failed to evaluate constraints at candidate point");
541 psi_new = this->merit(x_new, z_new, f, c, mu);
542
543 // Stop backtracking search if we've found a candidate point that sufficiently decreases
544 // the merit function and satisfies all the constraints
545 if ((c.array() <= 0.0).all() && psi_new < psi + this->m_tau * eta * alpha * dpsi) {
546 x = x_new;
547 z = z_new;
548 g_old = g;
549 break;
550 }
551
552 // The candidate point does not meet our criteria, so decrease the step size for 0 < < 1.
553 alpha *= this->m_beta;
554 if (alpha < this->m_alpha_min) {
555 IPSOLVER_ERROR(CMD "line search step size too small");
556 return false;
557 }
558 }
559 }
560 x_sol = x;
561 return converged;
562
563 #undef CMD
564 }

566private:
583 Real merit([[maybe_unused]] VectorN const & x, VectorM const & z, Real f, VectorM const & c,
584 Real const mu)
585 {
586 return f - c.dot(z) - mu * ((c.array().square() * z.array() + this->m_epsilon).log().sum());
587 }
588
604 Real grad_merit([[maybe_unused]] VectorN const & x, VectorM const & z, VectorN const & p_x,
605 VectorM const & p_z, VectorN const & g, VectorM const & c, MatrixJ const & J, Real const mu)
606 {
607 return p_x.dot(
608 g - J.transpose() * z - 2.0 * mu * J.transpose() * (1.0 / (c.array() - this->m_epsilon)).matrix()
609 ) - p_z.dot(
610 c + mu * (1.0 / (z.array() + this->m_epsilon)).matrix()
611 );
612 }

632 void bfgs_update(MatrixH & B, VectorN const & s, VectorN const & y)
633 {
634 IPSOLVER_ASSERT(y.dot(s) > 0.0,
635 "IPsolver::Solver::bfgs_update(...): update condition ys > 0 not satisfied");
636 VectorN x(B * s);
637 B -= (x * x.transpose()) / (s.dot(x)) + (y * y.transpose()) / (y.dot(s));
638 }
