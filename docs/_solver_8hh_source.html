259 Vector solve(const Vector& x_guess)
260 {
261 Vector x(x_guess);
262 Vector c(this->m_constraints(x));
263 Matrix J, W;
264 Integer n{static_cast<Integer>(x.size())};
265 Integer m{static_cast<Integer>(c.size())};
266 Integer nv{n + m};
267
268 Vector z(Vector::Ones(m));
269 Matrix B(Matrix::Identity(n, n));
270
271 Vector g_old;
272 Vector p_x;
273
274 if (this->m_verbose) {
275 std::cout << "i, f(x), lg(mu), sigma, ||r_x||, ||r_c||, alpha, #ls" << std::endl;
276 }
277
278 Real alpha{0.0};
279 Integer ls{0};
280
281 for (Integer iter{0}; iter < m_max_iterations; ++iter) {
282 Real f{this->m_objective(x)};
283 c = this->m_constraints(x);
284 J = this->m_constraints_jacobian(x, z);
285 W = this->m_lagrangian_hessian(x, z);
286
287 Vector g;
288 if (this->m_descent == Descent::NEWTON) {
289 g = this->m_objective_gradient(x);
290 B = this->m_objective_hessian(x);
291 } else {
292 g = this->m_objective_gradient(x);
293 }
294
295 Vector r_x{g + J.transpose() * z};
296 Vector r_c{c.array() * z.array()};
297 Vector r_0(nv);
298 r_0 << r_x, r_c;
299
300 Real norm_r0{static_cast<Real>(r_0.norm())};
301
302 Real eta{std::min(this->m_eta_max, norm_r0 / nv)};
303 Real sigma{std::min(this->m_sigma_max, std::sqrt(norm_r0 / nv))};
304 Real duality_gap{static_cast<Real>(-c.dot(z))};
305 Real mu{std::max(this->m_mu_min, sigma * duality_gap / m)};
306
307 if (this->m_verbose) {
308 std::cout << iter+1 << ", " << f << ", " << std::log10(mu) << ", " << sigma << ", "
309 << r_x.norm() << ", " << r_c.norm() << ", " << alpha << ", " << ls << std::endl;
310 }
311
312 if (norm_r0 / nv < this->m_tolerance) {break;}
313
314 if (this->m_descent == Descent::BFGS && iter > 1) {
315 Vector y(g - g_old);
316 B = this->bfgs_update(B, alpha * p_x, y);
317 }
318
319 Vector c_epsilon(c.array() - this->m_epsilon);
320 Matrix S(z.array() / c_epsilon.array());
321 Matrix S_diag(S.asDiagonal());
322
323 Vector g_b(g - mu * J.transpose() * (1.0 / c_epsilon.array()).matrix());
324
325 // Solve (B + W - J' * S * J) * px = -gb
326 Matrix H(B + W - J.transpose() * S_diag * J);
327 p_x = H.ldlt().solve(-g_b);
328
329 Vector p_z(-(z + mu * (1.0 / c_epsilon.array()).matrix() + S_diag * J * p_x));
330
331 alpha = this->m_alpha_max;
332 for (Integer i{0}; i < m; ++i) {
333 if (p_z[i] < 0) {alpha = std::min<Real>(alpha, this->m_alpha_max * (z[i] / (-p_z[i])));}
334 }
335
336 Real psi{this->merit(z, f, c, mu)};
337 Real dpsi{this->grad_merit(z, p_x, p_z, g, c, J, mu)};
338 ls = 0;
339
340 while (true) {
341 ls++;
342 Vector x_new(x + alpha * p_x);
343 Vector z_new(z + alpha * p_z);
344 Real f_new{this->m_objective(x_new)};
345 Vector c_new(this->m_constraints(x_new));
346 Real psi_new{this->merit(z_new, f_new, c_new, mu)};
347
348 bool allFeasible{(c_new.array() <= 0.0).all()};
349 if (allFeasible && psi_new < psi + this->m_tau * eta * alpha * dpsi) {
350 x = x_new;
351 z = z_new;
352 g_old = g;
353 break;
354 }
355 alpha *= this->m_beta;
356 if (alpha < this->m_alpha_min) {
357 IPSOLVER_ERROR("IPsolver::Solver::solve(...): line search step size too small");
358 }
359 }
360 }
361 return x;
362 }

380 Real merit(const Vector& z, Real f, const Vector& c, Real mu)
381 {
382 return f - c.dot(z) - mu * ((c.array().square() * z.array() + this->m_epsilon).log().sum());
383 }
384
399 Real grad_merit(const Vector& z, const Vector& p_x, const Vector& p_z, const Vector& g, const Vector& c,
400 const Matrix& J, Real mu)
401 {
402 Vector term1(g - J.transpose() * z - 2.0 * mu * J.transpose() * (1.0 / (c.array() - this->m_epsilon)).matrix());
403 Vector term2(c + mu * (1.0 / (z.array() + this->m_epsilon)).matrix());
404 return p_x.dot(term1) - p_z.dot(term2);
405 }

425 Matrix bfgs_update(const Matrix& B, const Vector& s, const Vector& y)
426 {
427 if (y.dot(s) <= 0) {IPSOLVER_ERROR("BFGS update condition ys > 0 not satisfied");}
428 Vector Bs(B * s);
429 return B - (Bs * Bs.transpose()) / (s.dot(Bs)) + (y * y.transpose()) / (y.dot(s));
430 }
