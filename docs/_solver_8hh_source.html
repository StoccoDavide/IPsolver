425 bool solve(const VectorN & x_guess, VectorN & x_sol)
426 {
427 #define CMD "IPsolver::Solver::solve(...): "
428
429 using MatrixM = Eigen::Matrix<Real, M, M>;
430 using ArrayM = Eigen::Array<Real, M, 1>;
431 using MaskM = Eigen::Array<bool, M, 1>;
432
433 // INITIALIZATION
434 // Get the number of primal variables (n), the number of constraints (m), the total number of
435 // primal-dual optimization variables (nv), and initialize the Lagrange multipliers and the
436 // second-order information
437 VectorN x(x_guess);
438 VectorM c(this->m_problem->constraints(x));
439 Integer n{static_cast<Integer>(x.size())};
440 Integer m{static_cast<Integer>(c.size())};
441 Integer nv{n + m};
442 VectorM z(VectorM::Ones(m));
443 MatrixH B(MatrixH::Identity(n, n));
444
445 // Repeat while the convergence criterion has not been satisfied, and we haven't reached the
446 // maximum number of iterations
447 Real alpha{0.0}, f, norm_r0, eta, sigma, duality_gap, mu, psi, dpsi, psi_new;
448 VectorN r_x, g_b, g_old, p_x, g, x_new;
449 VectorM r_c, c_epsilon, p_z, z_new;
450 MatrixJ J;
451 MatrixH H, W;
452 MatrixM S;
453 MaskM mask;
454 ArrayM ratio;
455 bool converged{false};
456 for (Integer iter{0}; iter < this->m_max_iterations; ++iter)
457 {
458
459 // COMPUTE OBJECTIVE, GRADIENT, CONSTRAINTS, ETC
460 // Compute the response of the objective function, the gradient of the objective, the
461 // response of the inequality constraints, the Jacobian of the inequality constraints, the
462 // Hessian of the Lagrangian (minus the Hessian of the objective) and, optionally, the
463 // Hessian of the objective.
464 f = this->m_problem->objective(x);
465 c = this->m_problem->constraints(x);
466 g = this->m_problem->objective_gradient(x);
467 J = this->m_problem->constraints_jacobian(x, z);
468 W = this->m_problem->lagrangian_hessian(x, z);
469 if (this->m_descent == Descent::NEWTON) {
470 B = this->m_problem->objective_hessian(x);
471 }
472
473 // Compute the responses of the unperturbed Karush-Kuhn-Tucker optimality conditions.
474 r_x = g + J.transpose() * z; // Dual residual
475 r_c = c.array() * z.array(); // Complementarity
476
477 // Set some parameters that affect convergence of the primal-dual interior-point method
478 norm_r0 = std::sqrt<Real>(r_x.squaredNorm() + r_c.squaredNorm());
479 eta = std::min<Real>(this->m_eta_max, norm_r0 / nv);
480 sigma = std::min<Real>(this->m_sigma_max, std::sqrt(norm_r0 / nv));
481 duality_gap = static_cast<Real>(-c.dot(z));
482 mu = std::max<Real>(this->m_mu_min, sigma * duality_gap / m);
483
484 if (this->m_verbose) {
485 std::cout << "Iteration " << iter << std::setprecision(6) << std::scientific
486 << ": f = " << f << ", |r| = " << norm_r0 / nv << std::endl;
487 }
488
489 // CONVERGENCE CHECK
490 // If the norm of the responses is less than the specified tolerance, we are done
491 if (norm_r0 / nv < this->m_tolerance) {
492 converged = true;
493 break;
494 }
495
496 // Update the BFGS approximation to the Hessian of the objective
497 if (this->m_descent == Descent::BFGS && iter > 0) {
498 B = this->bfgs_update(B, alpha * p_x, g - g_old);
499 }
500
501 // SOLUTION TO PERTURBED KKT SYSTEM
502 // Compute the search direction of x and z
503 c_epsilon = c.array() - this->m_epsilon;
504 S = (z.array() / c_epsilon.array()).matrix().asDiagonal();
505 g_b = g - mu * J.transpose() * (1.0 / c_epsilon.array()).matrix();
506 H = B + W - J.transpose() * S * J;
507 p_x = H.ldlt().solve(-g_b);
508 p_z = -(z + mu * (1.0 / c_epsilon.array()).matrix() + S * J * p_x);
509
510 // BACKTRACKING LINE SEARCH
511 // To ensure global convergence, execute backtracking line search to determine the step length
512 // First, we have to find the largest step size which ensures that z remains feasible
513 // Next, we perform backtracking line search
514 alpha = this->m_alpha_max;
515 mask = (z + p_z).array() < 0.0;
516 if (mask.any()) {
517 ratio = z.array() / (-p_z.array());
518 alpha = this->m_alpha_max * std::min<Real>(1.0, mask.select(ratio, 1.0).minCoeff());
519 }
520
521 // Compute the response of the merit function and the directional gradient at the current
522 // point and search direction
523 psi = this->merit(x, z, f, c, mu);
524 dpsi = this->grad_merit(x, z, p_x, p_z, g, c, J, mu);
525 while (true) {
526
527 // Compute the candidate point, the constraints, and the response of the objective function
528 // and merit function at the candidate point
529 x_new = x + alpha * p_x;
530 z_new = z + alpha * p_z;
531 f = this->m_problem->objective(x_new);
532 c = this->m_problem->constraints(x_new);
533 psi_new = this->merit(x_new, z_new, f, c, mu);
534
535 // Stop backtracking search if we've found a candidate point that sufficiently decreases
536 // the merit function and satisfies all the constraints
537 if ((c.array() <= 0.0).all() && psi_new < psi + this->m_tau * eta * alpha * dpsi) {
538 x = x_new;
539 z = z_new;
540 g_old = g;
541 break;
542 }
543
544 // The candidate point does not meet our criteria, so decrease the step size for 0 < < 1.
545 alpha *= this->m_beta;
546 if (alpha < this->m_alpha_min) {
547 IPSOLVER_ERROR(CMD "line search step size too small");
548 return false;
549 }
550 }
551 }
552 x_sol = x;
553 return converged;
554
555 #undef CMD
556 }
557
558private:
575 Real merit([[maybe_unused]] VectorN const & x, VectorM const & z, Real f, VectorM const & c,
576 Real const mu)
577 {
578 return f - c.dot(z) - mu * ((c.array().square() * z.array() + this->m_epsilon).log().sum());
579 }
580
596 Real grad_merit([[maybe_unused]] VectorN const & x, VectorM const & z, VectorN const & p_x,
597 VectorM const & p_z, VectorN const & g, VectorM const & c, MatrixJ const & J, Real const mu)
598 {
599 return p_x.dot(
600 g - J.transpose() * z - 2.0 * mu * J.transpose() * (1.0 / (c.array() - this->m_epsilon)).matrix()
601 ) - p_z.dot(
602 c + mu * (1.0 / (z.array() + this->m_epsilon)).matrix()
603 );
604 }
605
624 MatrixH bfgs_update(MatrixH const & B, VectorN const & s, VectorN const & y)
625 {
626 IPSOLVER_ASSERT(y.dot(s) > 0.0,
627 "IPsolver::Solver::bfgs_update(...): update condition ys > 0 not satisfied");
628 VectorN x(B * s);
629 return B - (x * x.transpose()) / (s.dot(x)) + (y * y.transpose()) / (y.dot(s));
630 }
